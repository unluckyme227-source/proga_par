using namespace std;

int main() {
    int n;
    cin >> n;

    int temp = n;
    int hasEven = 0;
    int product = 1;
    int sum = 0;

    __asm {
        mov eax, temp

        check_loop :
        cmp eax, 0
            je check_end

            mov edx, 0
            mov esi, 10
            div esi

            mov ecx, edx

            mov ebx, ecx
            mov eax, ebx
            mov edx, 0
            mov esi, 2
            div esi

            cmp edx, 0
            je even_found

            mov ebx, product
            imul ebx, ecx
            mov product, ebx

            mov eax, temp
            mov edx, 0
            mov esi, 10
            div esi
            mov temp, eax

            mov eax, temp
            jmp check_loop

            even_found :
        mov hasEven, 1

            mov eax, temp
            mov edx, 0
            mov esi, 10
            div esi
            mov temp, eax

            mov eax, temp
            jmp check_loop

            check_end :
    }

    if (hasEven) {
        cout << product;
    }
    else {
        temp = n;

        __asm {
            mov eax, temp

            sum_loop :
            cmp eax, 0
                je sum_end

                mov edx, 0
                mov esi, 10
                div esi

                add sum, edx

                mov temp, eax
                mov eax, temp

                jmp sum_loop

                sum_end :
        }

        cout << sum;
    }
        
    return 0;
}

#include <mpi.h>
#include <iostream>

using namespace std;

const int N = 15;  // размер матрицы

int main(int argc, char** argv) {
    MPI_Init(&argc, &argv);

    int rank, size;
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    int rows_per_proc = N / size;   // предполагаем, что N делится на size

    int* matrix = nullptr;          // полная матрица (только у root)
    int* local_matrix = new int[rows_per_proc * N]; // блок строк

    // root создаёт матрицу
    if (rank == 0) {
        matrix = new int[N * N];

        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                matrix[i * N + j] = i * N + j + 1;
            }
        }
    }

    MPI_Barrier(MPI_COMM_WORLD);
    double start_time = MPI_Wtime();

    // распределяем строки
    MPI_Scatter(
        matrix,
        rows_per_proc * N,
        MPI_INT,
        local_matrix,
        rows_per_proc * N,
        MPI_INT,
        0,
        MPI_COMM_WORLD
    );

    // каждый процесс считает свою часть диагонали
    int local_sum = 0;

    int start_row = rank * rows_per_proc;

    for (int i = 0; i < rows_per_proc; i++) {
        int global_i = start_row + i;

        if (global_i < N) {
            local_sum += local_matrix[i * N + global_i];
        }
    }

    int global_sum = 0;

    MPI_Reduce(&local_sum, &global_sum, 1, MPI_INT, MPI_SUM, 0, MPI_COMM_WORLD);

    double end_time = MPI_Wtime();

    if (rank == 0) {
        cout << "Trace sum = " << global_sum << endl;
        cout << "Time: " << end_time - start_time << " seconds" << endl;
    }

    // очистка памяти
    delete[] local_matrix;
    if (rank == 0) delete[] matrix;

    MPI_Finalize();
    return 0;
}

#include <iostream>
#include <vector>
#include <cmath>
#include <omp.h>

using namespace std;

const int N = 1000;
const double eps = 0.000001;
const int max_iter = 10000;

double jacobi_seq(vector<vector<double>>& A, vector<double>& b, vector<double>& x)
{
    vector<double> x_new(N, 0);

    double start = omp_get_wtime();

    for (int iter = 0; iter < max_iter; iter++)
    {
        double max_diff = 0;

        for (int i = 0; i < N; i++)
        {
            double s = 0;

            for (int j = 0; j < N; j++)
            {
                if (i != j)
                    s += A[i][j] * x[j];
            }

            x_new[i] = (b[i] - s) / A[i][i];
        }

        for (int i = 0; i < N; i++)
        {
            max_diff = max(max_diff, fabs(x_new[i] - x[i]));
            x[i] = x_new[i];
        }

        if (max_diff < eps)
            break;
    }

    double end = omp_get_wtime();

    return end - start;
}

double jacobi_parallel(vector<vector<double>>& A,
    vector<double>& b,
    vector<double>& x,
    int threads)
{
    vector<double> x_new(N, 0);

    omp_set_num_threads(threads);

    double start = omp_get_wtime();

    for (int iter = 0; iter < max_iter; iter++)
    {
        double max_diff = 0;

#pragma omp parallel for
        for (int i = 0; i < N; i++)
        {
            double s = 0;

            for (int j = 0; j < N; j++)
            {
                if (i != j)
                    s += A[i][j] * x[j];
            }

            x_new[i] = (b[i] - s) / A[i][i];
        }

        for (int i = 0; i < N; i++)
        {
            max_diff = max(max_diff, fabs(x_new[i] - x[i]));
            x[i] = x_new[i];
        }

        if (max_diff < eps)
            break;
    }

    double end = omp_get_wtime();

    return end - start;
}

int main()
{
    vector<vector<double>> A(N, vector<double>(N, 1));
    vector<double> b(N, N + 1);

    for (int i = 0; i < N; i++)
    {
        A[i][i] = 2 * N;
    }

    vector<double> x1(N, 0);

    double t1 = jacobi_seq(A, b, x1);

    cout << "seq time: " << t1 << endl << endl;

    int arr[] = { 2,4,8 };

    for (int k = 0; k < 3; k++)
    {
        int threads = arr[k];

        vector<double> x2(N, 0);

        double t2 = jacobi_parallel(A, b, x2, threads);

        double speedup = t1 / t2;
        double eff = speedup / threads;

        cout << "threads: " << threads << endl;
        cout << "time: " << t2 << endl;
        cout << "speedup: " << speedup << endl;
        cout << "efficiency: " << eff << endl;
        cout << endl;
    }

    return 0;
}

import java.io.*;
import java.util.*;

class MyThread extends Thread {

    private String fileName;

    public MyThread(String fileName) {
        this.fileName = fileName;
    }

    @Override
    public void run() {

        double mult = 1;
        boolean hasPositive = false;

        try {
            Scanner sc = new Scanner(new File(fileName));

            while (sc.hasNextDouble()) {

                double num = sc.nextDouble();

                if (num > 0) {
                    mult *= num;
                    hasPositive = true;
                }
            }

            sc.close();

            String newFile = fileName + "_res.txt";

            PrintWriter out = new PrintWriter(newFile);

            if (hasPositive)
                out.println(mult);
            else
                out.println("No positive numbers");

            out.close();

            System.out.println(fileName + " done");

        }
        catch (Exception e) {
            System.out.println("Error in file " + fileName);
        }
    }
}

public class Main {

    public static void main(String[] args) {

        String[] files = {
                "file1.txt",
                "file2.txt",
                "file3.txt"
        };

        MyThread[] threads = new MyThread[files.length];

        for (int i = 0; i < files.length; i++) {

            threads[i] = new MyThread(files[i]);
            threads[i].start();
        }

        for (int i = 0; i < files.length; i++) {

            try {
                threads[i].join();
            }
            catch (InterruptedException e) {
                e.printStackTrace();
            }
        }

        System.out.println("All threads finished");
    }
}

import java.io.*;
import java.util.Scanner;
import java.util.concurrent.*;

public class FileProductCalculator {
    
    public static void main(String[] args) {
        // Имена исходных файлов
        String[] inputFiles = {"data1.txt", "data2.txt", "data3.txt"};
        
        // Создаем потоки для каждого файла
        Thread[] threads = new Thread[inputFiles.length];
        
        for (int i = 0; i < inputFiles.length; i++) {
            final String inputFile = inputFiles[i];
            final String outputFile = inputFile + "_res";
            
            threads[i] = new Thread(() -> {
                processFile(inputFile, outputFile);
            });
            threads[i].start();
        }
        
        // Ждем завершения всех потоков
        for (Thread thread : threads) {
            try {
                thread.join();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
        
        System.out.println("Все файлы обработаны!");
    }
    
    private static void processFile(String inputFile, String outputFile) {
        double product = 1.0;
        boolean hasPositive = false;
        
        try (Scanner scanner = new Scanner(new File(inputFile));
             PrintWriter writer = new PrintWriter(new File(outputFile))) {
            
            while (scanner.hasNextDouble()) {
                double number = scanner.nextDouble();
                if (number > 0) {
                    product *= number;
                    hasPositive = true;
                    System.out.println(Thread.currentThread().getName() + 
                                     ": умножил " + number);
                }
            }
            
            // Записываем результат
            if (hasPositive) {
                writer.println("Произведение положительных элементов: " + product);
            } else {
                writer.println("Положительных элементов не найдено");
            }
            
            System.out.println(Thread.currentThread().getName() + 
                             ": файл " + inputFile + " обработан");
            
        } catch (FileNotFoundException e) {
            System.err.println(Thread.currentThread().getName() + 
                             ": файл " + inputFile + " не найден");
        }
    }
}
